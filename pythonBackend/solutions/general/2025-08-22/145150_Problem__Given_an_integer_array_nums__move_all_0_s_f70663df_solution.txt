function moveZeroes(nums) {
    // This function moves all zeros in the array to the end while maintaining the order of non-zero elements
    // Initialize a pointer for the position to place the next non-zero element
    let lastNonZeroFoundAt = 0; // Tracks the index for placing non-zero elements
    
    // Iterate through the array
    for (let i = 0; i < nums.length; i++) {
        // If the current element is not zero, we need to move it to the 'lastNonZeroFoundAt' position
        if (nums[i] !== 0) {
            // Swap elements at 'i' and 'lastNonZeroFoundAt' if they are different
            // This ensures that all non-zero elements are moved to the front
            [nums[lastNonZeroFoundAt], nums[i]] = [nums[i], nums[lastNonZeroFoundAt]]; // Swap operation
            lastNonZeroFoundAt++; // Move the pointer to the next position
        }
    }
    // The array is modified in place, so no return is necessary
}
